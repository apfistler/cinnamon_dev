{# tag.jinja #} {%- macro render_tag(type, tag_data) -%} {# Generate a single tag based on type and tag_data #} <{{ type }} {%- for tag in tag_data -%} {%- for key, value in tag.items() -%} {{ key }}="{{ value }}" {%- endfor -%} {%- endfor -%}> {%- endmacro -%} {%- macro render_tags(type, tags) -%} {# Generate multiple tags based on type and tags array #} {%- for tag_data in tags -%} {{ render_tag(type, tag_data) }} {%- endfor -%} {%- endmacro -%} {{ render_tags(type, values) }}
